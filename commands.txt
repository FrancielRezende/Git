#install git
winget install --id Git.Git -e --source winget

#check git version
git --version

echo "# Git" >> README.md
git init
git add README.md
git commit -m "first commit"
git branch -M main
git remote add origin https://github.com/FrancielRezende/Git.git
git push -u origin main

#config git user
git config --global user.name "francielrezende"

#config git user email
git config --global user.email "francielrezende@gmail.com"

#get git configs
git config user.name
git config user.email

#start a git directory
cd /local/forlder
git init

#check git status
git status

#track file / track all the project
git add file
git add .

#commit file with message / commit the hole repo
git commit -m file "message"
